DROP TABLE CUSTOMERS CASCADE CONSTRAINTS;

CREATE TABLE CUSTOMERS

(CUSTID INTEGER PRIMARY KEY NOT NULL,

CUSTFN VARCHAR2(20) NOT NULL,

CUSTLN VARCHAR2(20),

CUSTPHONE VARCHAR2(14) NOT NULL,

CUSTEMAIL VARCHAR2(30) NOT NULL,

CUSTSTREET VARCHAR2(30) NOT NULL,

CUSTCITY VARCHAR2(30) NOT NULL,

CUSTSTATE VARCHAR2(30) NOT NULL,

CUSTZIP NUMBER(5) NOT NULL);






DROP TABLE CREDITCARD CASCADE CONSTRAINTS;

CREATE TABLE CREDITCARD

(CCNO VARCHAR2(20) PRIMARY KEY NOT NULL,

EXPDATE VARCHAR2(7) NOT NULL,

CCFN VARCHAR2(30) NOT NULL,

CCLN VARCHAR2(30) NOT NULL,

CCSTREET VARCHAR2(30) NOT NULL,

CCCITY VARCHAR2(30) NOT NULL,

CCSTATE VARCHAR2(30) NOT NULL,

CCZIP NUMBER(5) NOT NULL, 

CUSTID INTEGER REFERENCES CUSTOMERS NOT NULL);






DROP TABLE CUSTORDER CASCADE CONSTRAINTS;

CREATE TABLE CUSTORDER

(CUSTORDID INTEGER PRIMARY KEY NOT NULL,

CUST_ORDERDATE DATE NOT NULL,

CUSTID INTEGER REFERENCES CUSTOMERS NOT NULL,

CCNO VARCHAR2(19) REFERENCES CREDITCARD NOT NULL);




DROP TABLE PRODUCTS CASCADE CONSTRAINTS;

CREATE TABLE PRODUCTS

(PRODUCTID INTEGER PRIMARY KEY NOT NULL,

DESCRIPTION VARCHAR2(100) NOT NULL,

SIZEE VARCHAR2(10) NOT NULL,

COLOR VARCHAR2(10) NOT NULL,

UNITPRICE DECIMAL(4,2) NOT NULL);



DROP TABLE CUSTORDERLINE CASCADE CONSTRAINTS;

CREATE TABLE CUSTORDERLINE

(CUSTOLID INTEGER PRIMARY KEY NOT NULL,

QUANTITY NUMBER(3) NOT NULL,

CUSTORDID INTEGER REFERENCES CUSTORDER NOT NULL,

PRODUCTID INTEGER REFERENCES PRODUCTS NOT NULL);



DROP TABLE SUPPLIERS CASCADE CONSTRAINTS;

CREATE TABLE SUPPLIERS

(SUPID INTEGER PRIMARY KEY NOT NULL,

SUPNAME VARCHAR2(30) NOT NULL,

SUPPHONE VARCHAR2(14) NOT NULL,

SUPEMAIL VARCHAR2(30) NOT NULL,

SUPSTREET VARCHAR2(30) NOT NULL,

SUPCITY VARCHAR2(30) NOT NULL,

SUPSTATE VARCHAR2(30) NOT NULL,

SUPZIP NUMBER(5) NOT NULL,

SUPWEBSITE VARCHAR2(60) NOT NULL);





DROP TABLE SUPORDER CASCADE CONSTRAINTS;

CREATE TABLE SUPORDER

(SUPORDID INTEGER PRIMARY KEY NOT NULL,

SUPID INTEGER REFERENCES SUPPLIERS NOT NULL);
